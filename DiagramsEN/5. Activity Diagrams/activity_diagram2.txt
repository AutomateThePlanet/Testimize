'Activity Diagram #2: Internal Logic of the ABC Algorithm.

@startuml
skinparam shadowing false
skinparam roundcorner 0
skinparam linetype ortho
skinparam nodesep 80
skinparam ranksep 50

start

:GenerateInitialPopulation(parameters);
:Evaluation via EvaluatePopulation(initialPopulation);

repeat
    :EvaluatePopulation(currentPopulation);

    :SelectElitePopulation();
    :Form non-elite as evaluated - elite;

    if (EnableOnlookerSelection?) then
        :nonElite = MaintainDiversePopulationOnlookerSelection(...);
    else
        :nonElite remains unchanged;
    endif

    :MutatePopulation(evaluated, nonElite, ...);

    if (EnableScoutPhase? && iteration > stagnationThreshold) then
        :PerformScoutPhaseIfNeeded();
    endif

repeat while (currentGeneration < totalGenerations)

:LimitFinalPopulationBasedOnSelectionRatio();
:Return finalPopulation;

stop
@enduml
