interface IDataProviderStrategy {
    +GenerateTestValues(): List<TestValue>
}

' === Абстрактни стратегии ===
abstract class EquivalenceOnlyDataProviderStrategy {
    +GenerateTestValues(): List<TestValue>
}
EquivalenceOnlyDataProviderStrategy ..|> IDataProviderStrategy

abstract class "BoundaryCapableDataProviderStrategy<T>" as BoundaryCapableDataProviderStrategy_T {
    +GenerateTestValues(): List<TestValue>
    +GenerateBoundaryValues(): List<TestValue>
}
BoundaryCapableDataProviderStrategy_T --|> EquivalenceOnlyDataProviderStrategy

' === Конкретни стратегии ===
class EmailDataProviderStrategy {
    +GenerateTestValues(): List<TestValue>
    +GenerateBoundaryValues(): List<TestValue>
}

class TextDataProviderStrategy {
    +GenerateTestValues(): List<TestValue>
